## awk 打印文件内容

    awk "{print } " /proc/meminfo 

## awk 字符串截断

	echo "/aaa/bbb/ccc" | awk -F '/' '{print $NF}'

>>> ccc
	
	echo hello:world:sf | awk -F":" '{ print $1 $3}' 

>>> hellosf

	echo hello:world:sf | awk -F":" '{ print $1" " $3}' 

>>> hello sf


	echo hello:world:sf | awk -F":" '{ print "username: " $1 "\t\tuid:" $3 }'

>>> username: hello		uid:sf
